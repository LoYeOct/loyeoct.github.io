      # C#阿里云短信接口

      # 注册账号审核模板

      首先要去阿里云注册用户账号，在“产品与服务”里按阿里的流程提示做好短信授权的准备，这里就不一一描述了，需要注意的是阿里云发送短信需要在后台先提交审核“签名名称”和“发送模板”，这些内容必须是审核通过的才能作为参数用于接口方法调用。（之前也用过其他短信接口，有一种是自定义模板发送的，但是存在不稳定延迟问题，建议还是用审核模板的方式，自定义本身就存在内容合法问题，延迟要命。）多申请几个模板，发送时根据使用场景传对应的模板ID发送即可。

      # 引用SDK

      SDK下载地址，还得CV对应语言SDK的Demo代码

      要引用的名称空间库也可以直接在Nuget上搜“aliyun-net-sdk-core”

      using System;
using System.Collections.Generic;
using Aliyun.Acs.Core;
using Aliyun.Acs.Core.Profile;
using Aliyun.Acs.Core.Exceptions;
using Aliyun.Acs.Core.Http;
using Newtonsoft.Json.Linq;
using Newtonsoft.Json;

namespace LoYe.Web
{
    public class ALYunSendSms
    {
        public JObject send(string phone,string SignName,string TemplateCode,string code)
        {
            JObject retJson = new JObject();
            IClientProfile profile = DefaultProfile.GetProfile("cn-shenzhen", "<accessKeyId>", "<accessSecret>");//accessKeyId和accessSecret每个账号开通短信是服务是固定的，建议直接写在config文件里调用
            DefaultAcsClient client = new DefaultAcsClient(profile);
            CommonRequest request = new CommonRequest();
            request.Method = MethodType.POST;
            request.Domain = "dysmsapi.aliyuncs.com";
            request.Version = "2020-05-25";
            request.Action = "SendSms";
            // request.Protocol = ProtocolType.HTTP;
            request.AddQueryParameters("PhoneNumbers", "" + phone + "");
            request.AddQueryParameters("SignName", "" + SignName + "");
            request.AddQueryParameters("TemplateCode", "" + TemplateCode + "");//
            request.AddQueryParameters("TemplateParam", "{\"code\":\"" + code + "\"}");
            try
            {
                //返回{"Message":"OK","RequestId":"56CC4E7F-94A7-42C8-AC90-68A985B8AB7C","BizId":"709314789969209259^0","Code":"OK"}
                CommonResponse response = client.GetCommonResponse(request);
                string retStr = System.Text.Encoding.Default.GetString(response.HttpResponse.Content);
                retJson = (JObject)JsonConvert.DeserializeObject(retStr);
            }
            catch (ServerException e)
            {
                string jsonText = "{\"Message\":\"" + e.Message + "\",\"RequestId\":\"" + e.RequestId + "\",\"ErrorCode\":\"" + e.ErrorCode + "\",\"ErrorMessage\":\"" + e.ErrorMessage + "\"}";
                retJson = (JObject)JsonConvert.DeserializeObject(jsonText);
            }
            catch (ClientException e)
            {
                string jsonText = "{\"Message\":\"" + e.Message + "\",\"RequestId\":\"" + e.RequestId + "\",\"ErrorCode\":\"" + e.ErrorCode + "\",\"ErrorMessage\":\"" + e.ErrorMessage + "\"}";
                retJson = (JObject)JsonConvert.DeserializeObject(jsonText);
            }
            return retJson;
        }
    }
}
