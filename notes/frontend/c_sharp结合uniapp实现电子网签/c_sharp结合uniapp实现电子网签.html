        # c#结合uniapp实现电子网签

        近期在做一个APP端电子签名功能,在APP端签名生成图片发送到服务端处理后按既定的模板生成PDF完成电子网签。

        # 一、生成PDF基础模板

        利用itextsharp类库（主角）来生成PDF，对这个类库的认识暂时还比较少，印象中之前只用它做过几次图片处理以及PDF添加水印之类的，不过不重要，引入类库有了轮子用就完事了。推荐在NuGet上下载，不要以为才一个.dll文件就去手动添加引用，得养成习惯，少走点弯路。

        以下是完整的生成代码，传入对应的参数是可以直接跑的，看个人需求，如果只是要生成PDF那前半部分几句就搞定了，思路是先生成一个基础的文字模板后再以这个生成的PDF文件上做改动，新增图片后再生成一个新的PDF，说起下面的内容参数content需要注意的是，像合同这种模板规范格式的，建议先存到多行文本框里面调整好直接传参就行了， iTextSharp.text.Paragraph方法是支持生成内容空格和换行（也就是直接把例如word文档内容CV过去就基本上搞定了，人名或者日期这些动态的可以用一些特殊字符替换，这些都需要提前准备好），千万别一行一行傻傻的去调用这个方法像字符串拼接一样拼格式！！！也许看到这你心里想的是“不会吧不会吧，不会真有人那么傻去拼接吧？”抱歉了，还真有，我第一次做这个的时候还真有人跟我说只能这拼接，所以他一直觉得生成PDF很麻烦。。。WTF？？？告辞。

        using iTextSharp.text;
using iTextSharp.text.pdf;
 public string CreatePdf(string signPic, string personName,string content,string commonPic)
        {
           
                string retStr = "";
                try
                {
                signPic = System.Web.HttpContext.Current.Server.MapPath(signPic);//签名照
                commonPic = System.Web.HttpContext.Current.Server.MapPath(commonPic);//公章

                string PdfPic = System.Web.HttpContext.Current.Server.MapPath("/upfiles/labor/" + DateTime.Now.ToString("yyMMddHHmmss") + personName + ".pdf");//生成的PDF路径
                Document doc = new Document(PageSize.A4, 36f, 72f, 108f, 180f); 
                BaseFont bf = BaseFont.CreateFont(@"c:\windows\fonts\SIMHEI.TTF", BaseFont.IDENTITY_H, BaseFont.NOT_EMBEDDED);//打印字体，当然也可以换其他的，如果不设置字体只能打印出英文和数字，无法打印中文
                iTextSharp.text.Font f = new Font(bf, 9);
                PdfWriter writer = PdfWriter.GetInstance(doc, new FileStream(PdfPic, FileMode.Create));//调用PDF生成类，第一个参数为生成地址
                writer.InitialLeading = 30;
                //  PdfWriter.GetInstance(document, new FileStream(fileName, FileMode.Create));
                // 添加文档信息
                doc.AddTitle("PdfInfo");
                doc.AddSubject("Demo of PDFInfo");
                doc.AddKeywords("Info, PDF, Demo");
                doc.AddCreator("SetPdfInfoDemo");
                doc.AddAuthor(personName);
                doc.Open();
                doc.Add(new iTextSharp.text.Paragraph(content, f));//传入内容参数生成
                doc.Close();
                //到这里就已经生成了一个名为PdfPic参数的PDF，到这就算成功了，以下就看个人需求了，我还得在已生成的PDF基础上加签名和公章已生成新的完整版PDF。


                PdfReader pdfReader = new PdfReader(PdfPic);//读需要改动的pdf文件
                string pdfWrite = "/upfiles/labor/" + DateTime.Now.ToString("yyMMddHHmmss") + personName + "_img.pdf";//新的PDF路径名称
                //-----------公章s------------
                PdfStamper pdfStamper = new PdfStamper(pdfReader, new FileStream(System.Web.HttpContext.Current.Server.MapPath(pdfWrite), FileMode.Create, FileAccess.Write, FileShare.None));
                PdfContentByte pdfAddimg = pdfStamper.GetOverContent(2);//获取pdf需要修改的页面
                iTextSharp.text.Image img = iTextSharp.text.Image.GetInstance(commonPic);
                img.SetAbsolutePosition(doc.PageSize.Width - 600f, doc.PageSize.Height - 710f);//图片生成位置，第一个参数以x轴从左开始算，第二个以y轴由下往上算，多调试几次就行了
                img.Alignment = iTextSharp.text.Image.ALIGN_CENTER;
                img.ScaleToFit(200f, 200f);//生成一个200*200的容器来存放图片，相当于按比例放大缩小了
                pdfAddimg.AddImage(img);
                //----------公章E----------------

                //---------签名照S-------------
                iTextSharp.text.Image img2 = iTextSharp.text.Image.GetInstance(signPic);
                img2.SetAbsolutePosition(doc.PageSize.Width - 325f, doc.PageSize.Height - 620f);//图片生成位置，第一个参数以x轴从左开始算，第二个以y轴由下往上算
                img2.Alignment = iTextSharp.text.Image.ALIGN_CENTER;
                img2.ScaleToFit(200f, 100f);//生成一个200*100的容器来存放图片
                img2.Rotation = (float)Math.PI / 2;//旋转角度，跟直接写90F一个意思，结果是竖版的签名照往左边旋转了90°
                pdfAddimg.AddImage(img2);
                //---------签名照E-------------
                pdfStamper.Close();

                retStr = pdfWrite;
            }
            catch (Exception ex) 
            {
               //记录错误
            }
            return retStr;
        }

        # 二、APP上电子签名并上传到服务端

        经过第一步服务端处理已经完成，接下来就只需要客户端生成一张电子签名然后上传到服务端进行获取处理便大功告成了，利用canvans绘图来完成，生成绘图代码网上随便搜都有，下面也直接贴签名页面（真CV即用）

        这里有个开发的坑要提一下的就是res.tempFilePath这个返回的是临时图片路径，这个坑就坑在，当时我跑这个页面的时候用的是PC端uni自带的内置浏览器运行跑的，控制台输出的值是Base64的图片编码格式，所以当时直接在服务端接参进行Base64转换成图片保存操作，结合第一步生成PDF的步骤，一切的一切进行的都是那么的顺利，在PC端完成了所谓的“鼠标电签”生成PDF，完美！紧接着换成真机调试一跑，GG，生成失败，失败归功于res.tempFilePath，在真机跑的时候返回是临时图片路径，如果这时候再转成Base64传到服务器就感觉浪费资源了，因为本身这个编码内容就不小，转来转去有时候避免不了出错，所以后面由post传参到服务器改成了上传附件的uni.uploadFile方法，对应的服务端也需要更改，在这先贴出服务端接收附件上传的方法

                string picPath = "/upfiles/signature/" + DateTime.Now.ToString("yyMMdd")+ personId + personName + ".png";

<script>
let content = null;
let touchs = [];
export default {
	onLoad: function(options) {
		//获得Canvas的上下文
		content = uni.createCanvasContext('sign');
		//设置线的颜色
		content.setStrokeStyle('red');
		//设置线的宽度
		content.setLineWidth(3);
		//设置线两端端点样式更加圆润
		content.setLineCap('round');
		//设置两条线连接处更加圆润
		content.setLineJoin('round');

		// content.fillRect(0, 0, 750, 700)
		// content.draw()
	},
	methods: {
		navTo(url) {
				uni.navigateTo({
				url
				});
			},
		// 画布的触摸移动开始手势响应
		start: function(event) {
			// console.log("触摸开始" + event.changedTouches[0].x);
			// console.log("触摸开始" + event.changedTouches[0].y);
			//获取触摸开始的 x,y
			let point = {
				x: event.changedTouches[0].x,
				y: event.changedTouches[0].y
			};
			touchs.push(point);
		},
		// 画布的触摸移动手势响应
		move: function(e) {
			let point = {
				x: e.touches[0].x,
				y: e.touches[0].y
			};
			touchs.push(point);
			if (touchs.length >= 2) {
				this.draw(touchs);
			}
		},
		// 画布的触摸移动结束手势响应
		end: function(e) {
			console.log('触摸结束' + e);
			//清空轨迹数组
			for (let i = 0; i < touchs.length; i++) {
				touchs.pop();
			}
		},
		// 画布的触摸取消响应
		cancel: function(e) {
			console.log('触摸取消' + e);
		},
		// 画布的长按手势响应
		tap: function(e) {
			console.log('长按手势' + e);
		},
		error: function(e) {
			console.log('画布触摸错误' + e);
		},
		//绘制
		draw: function(touchs) {
			let point1 = touchs[0];
			let point2 = touchs[1];
			touchs.shift();
			content.moveTo(point1.x, point1.y);
			content.lineTo(point2.x, point2.y);
			content.stroke();
			content.draw(true);
		},
		//清除操作
		clearClick: function() {
			//清除画布
			content.clearRect(0, 0, 750, 700);

			content.draw(true);
		},
		//保存图片
		saveClick: function() {
			var that = this;
			uni.canvasToTempFilePath({
				canvasId: 'sign',
				success: function(res) {
					let imgArr = [];
					imgArr.push(res.tempFilePath);
					// uni.previewImage({

							      url : _PageComm.App.app_Server+'project_server.ashx?type=getsignature',
							      filePath: res.tempFilePath,

							       'unid':_PageComm.getUsersUnid()

};
